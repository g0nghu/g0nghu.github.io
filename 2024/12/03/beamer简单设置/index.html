\usetheme{Madrid}

\usepackage[T1]{fontenc}
\usepackage{palatino}
\usepackage{mathrsfs}
\usepackage{calligra}
\usepackage{bm}

\usepackage{amsmath,amsthm,amsfonts,amssymb}

\usepackage{extarrows}
\usepackage{booktabs}

\usepackage{hyperref}
\hypersetup{
    colorlinks=true,
    linkcolor=,
    filecolor=blue,
    urlcolor=blue,
    citecolor=cyan,
}

\graphicspath{

}
